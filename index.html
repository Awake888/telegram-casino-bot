WHERE id = ?', (bet, win, user_id) ) conn.commit() conn.close() return web.json_response({'number': num, 'win': win}) # —————————————————————————————————————— # БЛЭКДЖЕК # —————————————————————————————————————— def get_card(): return random.choice([2, 3, 4, 5, 6, 7, 8, 9, 10, 10, 10, 10, 11]) def calc_hand(hand): s = sum(hand) # Если перебор и есть туз (11) — уменьшаем на 10 aces = hand.count(11) while s > 21 and aces: s -= 10 aces -= 1 return s async def bj_start(request): data = await request.json() user_id = data.get('user_id') bet = int(data.get('bet', 0)) if not user_id or bet < 1: return web.json_response({'error': 'Неверные данные'}, status=400) conn = get_conn() cursor = conn.cursor() cursor.execute('SELECT balance FROM users WHERE id = ?', (user_id,)) row = cursor.fetchone() if not row: conn.close() return web.json_response({'error': 'Пользователь не найден'}, status=404) if row['balance'] < bet: conn.close() return web.json_response({'error': 'Недостаточно монет!'}, status=400) # Резервируем ставку сразу при старте cursor.execute('UPDATE users SET balance = balance - ? WHERE id = ?', (bet, user_id)) conn.commit() conn.close() p_hand = [get_card(), get_card()] d_hand = [get_card(), get_card()] active_games[str(user_id)] = {'bet': bet, 'p': p_hand, 'd': d_hand} g = active_games[str(user_id)] return web.json_response({ 'player_hand': g['p'], 'player_score': calc_hand(g['p']), 'dealer_hand': [g['d'][0]], # Показываем только первую карту дилера 'dealer_score': g['d'][0], 'status': 'playing' }) async def bj_action(request): data = await request.json() user_id = str(data.get('user_id')) action = data.get('action') g = active_games.get(user_id) if not g: return web.json_response({'error': 'Игра не найдена. Начни новую.'}, status=400) if action == 'hit': g['p'].append(get_card()) if calc_hand(g['p']) > 21: return await end_bj(user_id, 'lose', '💥 Перебор! Ты проиграл.') elif action == 'stand': # Дилер добирает до 17 while calc_hand(g['d']) < 17: g['d'].append(get_card()) p_score = calc_hand(g['p']) d_score = calc_hand(g['d']) if d_score > 21 or p_score > d_score: return await end_bj(user_id, 'win', '🏆 Ты выиграл!') elif p_score < d_score: return await end_bj(user_id, 'lose', '😔 Дилер выиграл.') else: return await end_bj(user_id, 'draw', '🤝 Ничья!') return web.json_response({ 'player_hand': g['p'], 'player_score': calc_hand(g['p']), 'dealer_hand': [g['d'][0]], 'dealer_score': g['d'][0], 'status': 'playing' }) async def end_bj(user_id, result, msg): g = active_games.pop(user_id) bet = g['bet'] # Ставка уже списана при старте, поэтому: # win — возвращаем ставку + выигрыш (bet*2) # draw — возвращаем только ставку (bet) # lose — ничего не возвращаем (уже списано) if result == 'win': payout = bet * 2 elif result == 'draw': payout = bet else: payout = 0 conn = get_conn() cursor = conn.cursor() if payout > 0: cursor.execute('UPDATE users SET balance = balance + ?
